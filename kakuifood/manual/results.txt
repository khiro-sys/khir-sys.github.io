        <div class="bg-white p-8 rounded-lg shadow-xl text-center max-w-sm mx-auto">
            <h3 class="text-lg font-bold mb-4">本当にすべての記録を削除しますか？</h3>
            <p class="text-sm text-gray-600 mb-6">この操作は元に戻せません。</p>
            <div class="flex justify-center gap-4">
                <button id="cancel-delete" class="bg-gray-300 text-gray-800 font-bold py-2 px-6 rounded-lg hover:bg-gray-400">
                    キャンセル
                </button>
                <button id="confirm-delete" class="bg-red-600 text-white font-bold py-2 px-6 rounded-lg hover:bg-red-700">

    <script>
        const resultsBody = document.getElementById('results-body');
        const noResults = document.getElementById('no-results');
        const clearBtn = document.getElementById('clear-btn');
        const confirmModal = document.getElementById('confirm-modal');
        const cancelDeleteBtn = document.getElementById('cancel-delete');
        const confirmDeleteBtn = document.getElementById('confirm-delete');

        function loadResults() {
            resultsBody.innerHTML = '';
            const results = JSON.parse(localStorage.getItem('haccpQuizResults') || '[]');
            
            if (results.length === 0) {
                noResults.classList.remove('hidden');
            } else {
                noResults.classList.add('hidden');
                // 新しい記録が上に来るように逆順で表示
                results.reverse().forEach(result => {
                    const row = document.createElement('tr');
                    row.className = 'bg-white border-b';
                    
                    const scoreHistoryText = result.history.join('点, ') + '点';

                    row.innerHTML = `
                        <td class="px-6 py-4 font-medium text-gray-900 whitespace-nowrap">${result.date}</td>
                        <td class="px-6 py-4">${result.name}</td>
                        <td class="px-6 py-4 text-center">${result.attempts}</td>
                        <td class="px-6 py-4">${scoreHistoryText}</td>
                    `;
                    resultsBody.appendChild(row);
                });
            }
        }

        clearBtn.addEventListener('click', () => {
            confirmModal.classList.remove('hidden');
        });

        cancelDeleteBtn.addEventListener('click', () => {
            confirmModal.classList.add('hidden');
        });
        
        confirmDeleteBtn.addEventListener('click', () => {
            localStorage.removeItem('haccpQuizResults');
            confirmModal.classList.add('hidden');
            loadResults();
        });

        // ページ読み込み時に結果をロード
        window.addEventListener('load', loadResults);
    </script>
